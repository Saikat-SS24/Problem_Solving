def countApplesAndOranges(s, t, a, b, apples, oranges):
    # Write your code here
    apple_count = 0
    orange_count = 0
    
    # Check apple positions
    for d in apples:
        if s <= a + d <= t:
            apple_count += 1
            
    # Check orange positions
    for d in oranges:
        if s <= b + d <= t:
            orange_count += 1
            
    print(apple_count)
    print(orange_count)
