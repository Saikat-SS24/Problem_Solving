def bonAppetit(bill, k, b):
    # Write your code here
    total_cost = sum(bill)
    not_eaten = bill[k]
    b_actual = (total_cost - not_eaten) // 2
    
    if b > b_actual:
        print(b - b_actual)
        
    elif b == b_actual:
        print("Bon Appetit")

if __name__ == '__main__':
